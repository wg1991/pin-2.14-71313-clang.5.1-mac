154 static XED_INLINE xed_address_width_enum_t
155 xed_state_get_address_width(const xed_state_t* p)
156 {
157  switch(xed_state_get_machine_mode(p)) {
158  case XED_MACHINE_MODE_LONG_64:
159  return XED_ADDRESS_WIDTH_64b;
160 
161  case XED_MACHINE_MODE_REAL_16:
162  /* should be 20b... but if you are working w/real mode then you're
163  going to have to deal with somehow. Could easily make this be
164  20b if anyone cares. */
165  return XED_ADDRESS_WIDTH_32b;
166 
167  case XED_MACHINE_MODE_LEGACY_32:
168  case XED_MACHINE_MODE_LONG_COMPAT_32:
169  return XED_ADDRESS_WIDTH_32b;
170  case XED_MACHINE_MODE_LEGACY_16:
171  case XED_MACHINE_MODE_LONG_COMPAT_16:
172  return XED_ADDRESS_WIDTH_16b;
173  default:
174  return XED_ADDRESS_WIDTH_INVALID;
175  }
176 }
